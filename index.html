O) total_range = H - L df['Doji'] = (body <= (total_range * 0.1)) lower_shadow = np.where(C > O, O - L, C - L) upper_shadow = np.where(C > O, H - C, H - O) df['Hammer'] = (lower_shadow >= (2 * body)) & (upper_shadow <= (total_range * 0.1)) & (body > 0) df['Shooting_Star'] = (upper_shadow >= (2 * body)) & (lower_shadow <= (total_range * 0.1)) & (body > 0) df['Bullish_Engulfing'] = (C > O) & (C.shift(1) < O.shift(1)) & (C >= O.shift(1)) & (O <= C.shift(1)) df['Bearish_Engulfing'] = (C < O) & (C.shift(1) > O.shift(1)) & (C <= O.shift(1)) & (O >= C.shift(1)) return df @app.route('/') def home(): return jsonify({"status": "Server is running perfectly", "message": "Go to /api/scan to get data"}) @app.route('/api/scan') def scan(): scanner_results = [] for ticker in NIFTY_STOCKS: try: # 15 मिनट का डेटा डाउनलोड करना df = yf.download(ticker, period="5d", interval="15m", progress=False) if df.empty or len(df) < 20: continue if isinstance(df.columns, pd.MultiIndex): df = df[ticker] # इंडिकेटर्स कैलकुलेशन df['EMA_9'] = df['Close'].ewm(span=9, adjust=False).mean() df['EMA_21'] = df['Close'].ewm(span=21, adjust=False).mean() df['VWAP'] = calculate_vwap(df) # ATR कैलकुलेशन (Stop Loss के लिए वोलैटिलिटी) high_low = df['High'] - df['Low'] high_close = np.abs(df['High'] - df['Close'].shift()) low_close = np.abs(df['Low'] - df['Close'].shift()) tr = pd.concat([high_low, high_close, low_close], axis=1).max(axis=1) df['ATR'] = tr.ewm(span=14, adjust=False).mean() # कैंडल पैटर्न्स डिटेक्ट करना df = detect_candle_patterns(df) # 15-Min Pivot Points आधारित Support & Resistance daily_data = yf.download(ticker, period="2d", interval="1d", progress=False) if isinstance(daily_data.columns, pd.MultiIndex): daily_data = daily_data[ticker] prev_day = daily_data.iloc[-2] P = float((prev_day['High'] + prev_day['Low'] + prev_day['Close']) / 3) R1 = float((2 * P) - prev_day['Low']) S1 = float((2 * P) - prev_day['High']) latest = df.iloc[-1] prev = df.iloc[-2] current_price = float(latest['Close']) vwap_val = float(latest['VWAP']) ema9 = float(latest['EMA_9']) current_vol = float(latest['Volume']) prev_vol = float(prev['Volume']) atr = float(latest['ATR']) # 💡 केवल वॉल्यूम बढ़ने की कंडीशन (Current Volume > Previous Volume) # यदि आप दोगुना (2x) चाहते हैं तो इसे: current_vol >= (prev_vol * 2) कर सकते हैं volume_condition = current_vol > prev_vol if volume_condition: # यह तय करने के लिए कि ट्रेंड बुलिश है या बेयरिश (ताकि टारगेट/SL सही बने) is_bullish = current_price > vwap_val and current_price > ema9 pattern = "Volume Boost" if latest['Hammer']: pattern = "Hammer Breakout 🔨" elif latest['Bullish_Engulfing']: pattern = "Bullish Engulfing 🔥" elif latest['Shooting_Star']: pattern = "Shooting Star 🌠" elif latest['Bearish_Engulfing']: pattern = "Bearish Engulfing 📉" # स्टॉप लॉस और टारगेट कैलकुलेशन (ट्रेंड के आधार पर) if is_bullish: stop_loss = min(current_price - (1.5 * atr), S1 - 0.5) target = current_price + (np.abs(current_price - stop_loss) * 2) order_flow = f"बायर्स मजबूत हैं (वॉल्यूम बूस्ट)" else: stop_loss = max(current_price + (1.5 * atr), vwap_val) target = current_price - (np.abs(stop_loss - current_price) * 2) order_flow = f"सेलर्स का दबाव है (वॉल्यूम बूस्ट)" scanner_results.append({ 'ticker': ticker.replace('.NS', ''), 'price': round(current_price, 2), 'volume_up': round(current_vol / prev_vol, 1), 'support': round(S1, 2), 'resistance': round(R1, 2), 'entry': round(current_price, 2), 'target': round(target, 2), 'sl': round(stop_loss, 2), 'order_flow': f"{pattern} | {order_flow}" }) except Exception as e: print(f"Error scanning {ticker}: {e}") return jsonify(scanner_results)
